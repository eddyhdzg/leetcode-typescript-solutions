# 150. Evaluate Reverse Polish Notation

### O(n) - TS with Stack

## Complexity

- Time: $$O(n)$$
- Space: $$O(n)$$

## Submission

Runtime: 7 ms · Beats 51.05%
Memory: 59.22 MB · Beats 60.11%

## Code

```typescript
const OPS = {
  "+": (a: number, b: number) => a + b,
  "-": (a: number, b: number) => a - b,
  "*": (a: number, b: number) => a * b,
  "/": (a: number, b: number) => Math.trunc(a / b),
} as const;

type Operator = keyof typeof OPS;
const isOperator = (t: string): t is Operator => t in OPS;

function evalRPN(tokens: string[]): number {
  const stack: number[] = [];

  for (const token of tokens) {
    if (isOperator(token)) {
      const b = stack.pop() ?? 0;
      const a = stack.pop() ?? 0;
      stack.push(OPS[token](a, b));
    } else stack.push(+token);
  }

  return stack[0];
}
```
