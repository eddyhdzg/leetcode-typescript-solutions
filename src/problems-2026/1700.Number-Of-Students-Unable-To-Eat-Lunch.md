# 1700. Number Of Students Unable To Eat Lunch

### O(n) - TS with counting

## Complexity

- Time: $$O(n)$$
- Space: $$O(1)$$

## Submission

## Code

```typescript
const SANDWICH = { CIRCULAR: 0, SQUARE: 1 } as const;

function countStudents(students: number[], sandwiches: number[]): number {
  let circular = 0;
  let square = 0;

  for (const student of students) {
    if (student === SANDWICH.SQUARE) square++;
    else circular++;
  }

  for (const sandwich of sandwiches) {
    if (sandwich === SANDWICH.SQUARE) {
      if (square === 0) break;
      square--;
    } else {
      if (circular === 0) break;
      circular--;
    }
  }

  return square + circular;
}
```
