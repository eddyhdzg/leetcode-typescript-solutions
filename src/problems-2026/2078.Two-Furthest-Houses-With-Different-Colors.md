# 2078. Two Furthest Houses With Different Colors

### O(n) - TS with two pointers

- Space: $$O(1)$$

```typescript
// The farthest pair must include either the first or last house.
// Check both ends, take the max.
function maxDistance(colors: number[]): number {
  let res = 0;

  // Fix left (0), scan from right → first mismatch is farthest
  for (let i = colors.length - 1; i >= 0; i--) {
    if (colors[0] !== colors[i]) {
      res = Math.max(res, i - 0);
      break;
    }
  }

  // Fix right (last), scan from left → first mismatch is farthest
  for (let i = 0; i < colors.length - 1; i++) {
    if (colors[colors.length - 1] !== colors[i]) {
      res = Math.max(res, colors.length - 1 - i);
      break;
    }
  }

  return res;
}
```
