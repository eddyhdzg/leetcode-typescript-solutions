# 2748. Number Of Beautiful Pairs

### O(n²) - TS with nested loops

## Complexity

- Time: $$O(n^2)$$
- Space: $$O(1)$$

## Submission

Runtime: 31 ms · Beats 41.67%
Memory: 59.14 MB · Beats 66.67%

## Code

```typescript
// Euclidean algorithm
// https://douglasmoura.dev/finding-the-greatest-common-divisor-in-typescript
const gcd = (a: number, b: number): number => (b === 0 ? a : gcd(b, a % b));

const getFirstDigit = (num: number): number => +String(num)[0];

const getLastDigit = (num: number): number => num % 10;

function countBeautifulPairs(nums: number[]): number {
  let res = 0;

  for (let i = 0; i < nums.length; i++) {
    for (let j = i + 1; j < nums.length; j++) {
      if (gcd(getFirstDigit(nums[i]), getLastDigit(nums[j])) === 1) res++;
    }
  }

  return res;
}
```
